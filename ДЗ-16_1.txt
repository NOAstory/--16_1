using System;

class Program
{
    static void Main()
    {
        const int MinDivisor = 10;
        const int MaxDivisor = 25;
        const int StartRange = 50;
        const int EndRange = 150;

        Random random = new Random();
        int divisor = random.Next(MinDivisor, MaxDivisor + 1);

        Console.WriteLine($"Делитель: {divisor}");

        int count = 0;

        int firstMultiple = StartRange;
        int remainderCheck = firstMultiple;

        while (remainderCheck >= divisor)
        {
            remainderCheck -= divisor;
        }

        if (remainderCheck > 0)
        {
            firstMultiple += divisor - remainderCheck;
        }

        for (int number = firstMultiple; number <= EndRange; number += divisor)
        {
            count++;
        }

        Console.WriteLine($"Количество чисел от {StartRange} до {EndRange} кратных {divisor}: {count}");
    }
}
